function changeH(str) {
var MsgPop=document.getElementById("winpop");
var popH=parseInt(MsgPop.style.height);
if(str=="up"){     //如果这个参数是UP

clearInterval(show);//否则就取消这个函数调用,意思就是如果高度超过59象度了,就不再增长了
