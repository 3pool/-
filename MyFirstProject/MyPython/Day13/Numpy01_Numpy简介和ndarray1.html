def pySum(n):
    # n是序列的长度
    a = [i**2 for i in range(n)]
    b = [i**3 for i
         in range(n)]
    c = []
    # 空列表，存储和
    for i in range(n):
        # 遍历列表的索引
        c.append(a[i] + b[i])
    return c

import numpy as np
def npSum(n):
    # n是数组的长度
    a = np.arange(n) ** 2
    b = np.arange(n) ** 3
    return a+b

# numpy采用的数据类型和python不同，默认的整数类型是int32，遇到大值会产生溢出，可以改成int64
def npSumBig(n):
    # n是数组的长度
    a = np.arange(n,dtype = np.int64) ** 2
    # 指定数组的数据类型是int64
    b = np.arange(n,dtype = np.int64) ** 3
    return a + b
